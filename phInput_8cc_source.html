225 static void readInputFile(
226  const char* filename,
227  StringMap& stringMap,
228  IntMap& intMap,
229  DblMap& dblMap,
230  pcu::PCU *PCUObj)
231 {
232  stringset old;
233  makeDeprecated(old);
234  std::ifstream f(filename);
235  if (!f)
236  fail("could not open \"%s\"", filename);
237  std::string name;
238  while (f >> name) {
239  if (name[0] == '#' || deprecated(old,name,PCUObj)) {
240  std::getline(f, name, '\n');
241  continue;
242  }
243  if (tryReading(name, f, stringMap))
244  continue;
245  if (tryReading(name, f, intMap))
246  continue;
247  if (tryReading(name, f, dblMap))
248  continue;
249  fail("unknown variable \"%s\" in %s\n", name.c_str(), filename);
250  }
251 }
